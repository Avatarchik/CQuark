	}
}

		break;

}

			break;
		UnityEngine.Debug.Log(i.ToString());
	}
}

	UnityEngine.Debug.Log("HH"+i);
	for(int j = 0; j < 10; j++){
		if(j<2)
			continue;
		if(i > 5)
			break;
		if((i+j) % 3 == 0){
			UnityEngine.Debug.Log("" + i + j);

	}
}
